import numpy as np

class ZipImage:
    def __init__(self, Z):
        '''
        Z: å›¾ç‰‡çš„ ZipFile å¯¹è±¡
        '''

        self.name_bunch = {name: name.split('/')[0] for name in self.Z.namelist()}

    def array(self, name):
        with self.Z.open(name) as fp:
            with Image.open(fp) as im:
                img = np.array(im)
        return img

    def __iter__(self):
        for name, label in self.name_bunch.items():
            img = self.array(name)
            yield img, label
